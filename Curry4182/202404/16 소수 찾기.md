class Solution {
    public int solution(int n) {
        int answer = 0;
        
        boolean[] visit = new boolean[n+1];
        
        for(int i=2; i<visit.length; i++) {
            for(int j=2; j*i < visit.length; j++) {
                visit[i*j] = true;
            }
        }
        
        for(int i=2; i<=n; i++) {
            if(visit[i]) {
                continue;
            }
            answer += 1;
        }
                
        return answer;
    }
}
