import java.util.*;

class Solution {
    
    Character[][] map;
    
    int[] dy = {-1, 0, 1, 0};
    int[] dx = {0, 1, 0, -1};
    
    int n, m;
    
    class Node {
        
        public int y;
        public int x;
        public int cnt;

        
        public Node(int y, int x, int cnt) {
            this.y = y;
            this.x = x;
            this.cnt = cnt; 
        }
    }
    
    public boolean isRange(int y, int x) {
        return -1 < y && y < n && -1 < x && x < m;
    }
    
    public int solution(String[] board) {
        int answer = 0;
        
        n = board.length;
        m = board[0].length();
        
        map = new Character[n][m];

        boolean[][] visit = new boolean[n][m];
        Queue<Node> que = new LinkedList<>();
        for(int i=0; i<n; i++) {
            Character[]  line = new Character[m];
            String str = board[i];
            for(int j=0; j<m; j++) {
                line[j] = str.charAt(j);
                
                if(line[j] == 'R') {
                    que.add(new Node(i, j, 0));
                    visit[i][j] = true;
                }
            }
            map[i] = line;
        }
        
        while(que.size() != 0) {
            
            var sNode = que.poll();
            
            int ny, nx;
            for(int i=0; i<4; i++) {
                int ty = sNode.y + dy[i];
                int tx = sNode.x + dx[i];
                
                if(!isRange(ty, tx)) {
                    continue;
                }
                
                if(map[ty][tx] == 'D') {
                    continue;
                }
                
                ty = sNode.y;
                tx = sNode.x;
                           
                while(true) {
                    ny = ty + dy[i];
                    nx = tx + dx[i];
                                        
                    if(!isRange(ny, nx)) {
                        break;
                    }

                    if(map[ny][nx] == 'D') {
                        break;
                    }
                    
                    ty = ny;
                    tx = nx;
                }
                
                if(visit[ty][tx]) {
                    continue;
                }
                
                if(map[ty][tx] == 'G') {
                    return sNode.cnt + 1;
                }
                
                visit[ty][tx] = true;
                
                que.add(new Node(ty, tx, sNode.cnt + 1));
            }           
        }        
        
        
        return -1;
    }
}
