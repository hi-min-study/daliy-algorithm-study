import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int N = sc.nextInt();
		int m = 1000000000;

		int[][] dp = new int[N+1][10];

		for(int i=1; i<=9; i++) {
			dp[1][i] = 1;
		}

		for(int i=2; i<=N; i++) {
			for(int j=0; j<10; j++) {
				if (j != 0) {
					dp[i][j] = (dp[i][j] + dp[i-1][j - 1]) % m;
				}
				if (j != 9) {
					dp[i][j] = (dp[i][j] + dp[i-1][j + 1]) % m;
				}
			}
		}

		int r = 0;
		for(int i=0; i<10; i++) {
			r = (r + dp[N][i]) % m;
		}

		System.out.println(r);
	}
}
